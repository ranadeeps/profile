/* Styling for all text/email/password input elements */

input[type="email"],

  padding: 10px 15px;
  margin: 8px 0;
  display: inline-block;
  box-sizing: border-box; /* Padding won't increase the total width */

  /* 2. Unfocused (Default) Border */

  border-radius: 4px; /* Slightly rounded corners */

  /* 3. Smooth Transition */

}

}
